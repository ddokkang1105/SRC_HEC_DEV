02 Create base model procedure query 수정

SELECT ST.ClassCode, ST.ToItemID,(select COUNT(ToItemID) from xboltadm.TB_ITEM where FromItemID = @RItemID And ItemTypeCode in ('CN00001' , 'CN00101') AND Deleted != 1) AS ItemCnt
			FROM XBOLTADM.TB_ITEM ST 
			left outer join XBOLTADM.TB_ITEM ITM on ITM.ItemID = ST.ToItemID		
			INNER join XBOLTADM.TB_ITEM L5 on L5.ItemID = ST.ToItemID and L5.ClassCode = 'CL01006'
			WHERE ST.FromItemID = @RItemID And ST.ItemTypeCode in ('CN00001' , 'CN00101') AND ST.Deleted != 1 		
			order by ClassCode DESC , ITM.Identifier
